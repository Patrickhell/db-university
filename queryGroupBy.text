SELECT COUNT(`id`),YEAR(`enrolment_date`) AS 'Anno di iscrizione'
FROM `students` 
GROUP BY `Anno di iscrizione` ;

Domanda 2:
SELECT COUNT(`id`), `teachers`.`office_address`
FROM `teachers` 
GROUP BY `office_address`;

Domanda 3:

FROM `exam_student` 
GROUP BY `exam_id`;
